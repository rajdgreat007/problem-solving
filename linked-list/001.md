class Node {
  constructor(data){
    this.data = data;
    this.next = null;
  }
}

class LinkedList {
  constructor(){
    this.head = null;
  }

  addNode(data) {
    const node = new Node(data);
    if(this.head === null){
      this.head = node;
      return;
    }

    let temp = this.head;
    while(temp.next !== null){
      temp = temp.next;
    }
    temp.next = node;
  }

  printNodes() {
    let temp = this.head, output = [];
    while(temp !== null){
      output.push(temp.data);
      temp = temp.next;
    }
    return output.join(",");
  }

  countNodes = () => {
    if(this.head === null) return 0;
    let temp = this.head, count = 0;
    while(temp !== null){
      count++;
      temp = temp.next;
    }
    return count;
  }

  //remove last node
  removeLastNode = () => {
    if(this.head === null) return;
    let temp = this.head, previous;
    while(temp.next !== null) {
      previous = temp;
      temp = temp.next;
    }
    previous.next = null;
  }

  deleteLinkedList = () => {
    if(this.head === null) return;
    this.head = null;
  }

  // Insert at the nth position in the list. 
  // Return if the number of nodes is less than n
  insertNodeAtIndex = (n, data) => {
    const length = this.countNodes();
    if(length < n || n < 0) return;
    if(this.head === null) this.addNode(data);
    const node = new Node(data);
    if(n === 0) {
      node.next = this.head;
      this.head = node;
      return;
    }
    let idx = 0, temp = this.head;
    while(temp !== null) {
      if(idx === n-1) {
        node.next = temp.next;
        temp.next = node;
        return;
      }
      idx++;
      temp = temp.next;
    }
  }
  
  insertSorted(data) {
    const node = new Node(data);
    if(this.head === null || data < this.head.data) {
      const temp = this.head;
      this.head = node;
      node.next = temp;
      return;
    }
    
    let curr = this.head;
    let next = curr.next;
    while(next !== null) {
      if( data <= next.data) {
        node.next = next;
        curr.next = node;
        return;
      }else {
        const temp = next;
        curr = next;
        next = temp.next;
      }
    }
    curr.next = node;
  }
  
  appendList(list){
    if(this.head === null) {
      this.head = list;
      return;
    }
    let temp = this.head;
    while(temp.next !== null) {
       temp = temp.next;
    }
    temp.next = list;
  }

  // Split a linked list into 2 equal parts. If there are an odd number of elements, the extra element should be in the first list.

  splitLinkedList(){
    if(this.head === null) return;
    let slow = this.head, fast = slow.next;
    const front = [slow.data], back = [];
    while(fast !== null && fast.next !== null) {
      slow = slow.next;
      front.push(slow.data);
      fast = fast.next && fast.next.next;
    }
    while(slow !== null){
      slow = slow.next;
      if(slow) back.push(slow.data);
    }
    return {front, back}
  }

 // remove duplicates from sorted list (keep only 1 item for each duplicate)

removeDuplicatesSortedList(){
  if(this.head === null || this.head.next === null) return;
  
  let curr = this.head;
  while(curr.next !== null) {
    if(curr.data === curr.next.data){
      curr.next = curr.next.next;
    } else {
      curr = curr.next;
    }
  }
}

// Given a sorted linked list, delete all nodes that have duplicate numbers (all occurrences), leaving only numbers that appear once in the original list.

  removeAllDuplicatesSortedList(){
    const dummy = new Node(0);
    dummy.next = this.head;
    let prev = dummy;
    let curr = this.head;
    
    while(curr !== null){
      while(curr.next !== null && prev.next.data === curr.next.data){
        curr = curr.next;
      }
      
      if(prev.next === curr) {
        prev = prev.next
      } else {
        prev.next = curr.next;
      }
      curr = curr.next;
    }
    this.head = dummy.next;
  }

//Create a new sorted list which is the merged from two original sorted lists.Assume the lists are sorted in ascending order.

mergeSortedLists(head1, head2) {
    let temp1 = head1, temp2 = head2;
    let list3 = {head:null};
    if(temp1 === null) return temp2;
    if(temp2 === null) return temp1;
    
    while(temp1 !== null && temp2 !== null) {
      if(temp1.data <= temp2.data) {
        this.addNode.call(list3, temp1.data);
        temp1 = temp1.next;
      } else {
        this.addNode.call(list3, temp2.data);
        temp2 = temp2.next;
      }
    }
    
    if(temp1 === null) {
      while(temp2 !== null) {
        this.addNode.call(list3, temp2.data);
        temp2 = temp2.next;
      }
    }
    
    if(temp2 === null) {
      while(temp1 !== null) {
        this.addNode.call(list3, temp1.data);
        temp1 = temp1.next;
      }
    }
    return list3;
  }

  //Reverse all the nodes in the linked list so that the last node becomes the first node
  reverseLinkedList(){
    if(this.head === null || this.head.next === null) return;
    let temp = this.head.next, prev = this.head, next;
    prev.next = null;
    while(temp !== null) {
      next = temp.next;
      temp.next = prev;
      prev = temp;
      temp = next;
    }
    this.head = prev;
  }

  reverseLinkedListRecursive(temp, prev) {
    if(temp === null) {
      this.head = prev;
      return;
    }
    
    this.reverseLinkedListRecursive(temp.next, temp);
    temp.next = prev;
    
  }
}
